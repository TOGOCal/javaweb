
insert into 表名 (列名1，列名2...) values (值1，值2...)
-- 值与列名意一一匹配
-- 如果需要对表中所有列添加信息，则需要将所有列名都输入一遍
-- 这非常麻烦，因此就有了下面这种写法


insert into 表名 values (值1 ，值2...)；
-- 给全部列添加数据（这是那种方法的简写

insert into 表名 
values 
(值1，值2...),
(值1，值2...),
(值1，值2...);
-- 可以通过这种方式使用同一个输入模板进行多条输入
-- 不仅是这种简写方式，最原始的那种也可以使用这种模式

update 表名 set 列名1 = 值1， 列名2 = 值2 ...[where 条件];
-- []中的内容就是可要可不要的
-- 如果不要where就代表将该列的所有数据修改
-- 但比如你知道A同学的id，你要将id为该id的学生姓名修改为A
-- 就需要
-- update 表名 set name = A where id = this.id;

delete from 表名 [where 条件]
-- 删除条件
